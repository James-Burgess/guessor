import requests

username = "your_username"  # Replace with the actual username

for a in range(ord('a'), ord('z') + 1):
    for b in range(ord('a'), ord('z') + 1):
        for c in range(ord('a'), ord('z') + 1):
            for d in range(ord('a'), ord('z') + 1):
                word = chr(a) + chr(b) + chr(c) + chr(d)
                full_url = base_url + word

                # Include the username cookie directly in the request headers
                response = requests.get(full_url, cookies={"username": username})

                if response.status_code == 200:
                    print("Success! The word is:", word)
                    break
